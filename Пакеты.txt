/etc/apt/sources.list - список реп

Внешние: 

    apt-get

        remoove --purge <имя пакета> - удаление пакета
        changelog <имя пакета> - изменения в пакете

    apt

Внутренние:

    apt

    apt-cash

        pkgnames - все доступные для установки пакеты

        search <имя пакета/программы> - если не знаем как точно называется прога

        policy <имя пакета> - посмотреть версии пакета
    
    dpkg
        
        dpkg -l - список устаглвденных пакетов
        dpkg -s <имя пакета> - инфо о пакете
        dpkg -L <имя пакета> - файлы пакета
        dpkg -i <файд пакета>.deb - установить скачанный пакет
        dpkg -r <файд пакета>.deb - удалить скачанный пакет
